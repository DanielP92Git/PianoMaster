```typescript
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';

const ALLOWED_ORIGINS = ['https://testpianomaster.netlify.app', 'http://localhost:5174'];

function getCorsHeaders(req: Request) {
  const origin = req.headers.get('origin') || '';
  return {
    'Access-Control-Allow-Origin': ALLOWED_ORIGINS.includes(origin) ? origin : ALLOWED_ORIGINS[0],
    'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
    'Access-Control-Allow-Methods': 'POST, OPTIONS',
  };
}

// jsonResponse helper (inside Deno.serve callback):
function jsonResponse(body: unknown, status = 200): Response {
  return new Response(JSON.stringify(body), {
    status,
    headers: { ...corsHeaders, 'Content-Type': 'application/json' },
  });
}

// JWT extraction:
const authHeader = req.headers.get('Authorization');
if (!authHeader) {
  return jsonResponse({ error: 'Unauthorized' }, 401);
}
const supabaseUser = createClient(
  Deno.env.get('SUPABASE_URL')!,
  Deno.env.get('SUPABASE_ANON_KEY')!,
  { global: { headers: { Authorization: authHeader } } }
);
const { data: { user }, error: authError } = await supabaseUser.auth.getUser();
```

```sql
-- Feedback submission tracking for rate limiting.
-- Message content is NOT stored — sent via Brevo only (per D-07, COPPA-safe).

CREATE TABLE IF NOT EXISTS feedback_submissions (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  student_id UUID NOT NULL REFERENCES students(id) ON DELETE CASCADE,
  type TEXT NOT NULL CHECK (type IN ('bug', 'suggestion', 'other')),
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Index for rate check query: COUNT(*) WHERE student_id = X AND created_at > 1 hour ago
CREATE INDEX idx_feedback_submissions_student_created
  ON feedback_submissions(student_id, created_at DESC);

ALTER TABLE feedback_submissions ENABLE ROW LEVEL SECURITY;

-- Per D-10: authenticated users can INSERT own rows only.
-- No SELECT/UPDATE/DELETE policies for clients.
-- Service role bypasses RLS for rate check COUNT query.
CREATE POLICY "Students can insert own feedback"
  ON feedback_submissions
  FOR INSERT
  TO authenticated
  WITH CHECK (student_id = auth.uid());

GRANT INSERT ON feedback_submissions TO authenticated;

COMMENT ON TABLE feedback_submissions IS
  'Tracks feedback form submissions for rate limiting. Message content is not stored — sent via Brevo only.';
```

Create `supabase/functions/send-feedback/index.ts` — a single-file Edge Function (self-contained, no shared modules) that composites patterns from `create-checkout` (JWT extraction, jsonResponse) and `send-consent-email` (CORS, Brevo API call with AbortController timeout).

**Full implementation specification:**

```typescript
/**
 * Send Feedback Edge Function
 *
 * Receives authenticated feedback submissions, validates input,
 * enforces rate limiting (3/hour per user), and sends plain-text
 * email to support inbox via Brevo API.
 *
 * Request Body:
 *   - type: 'bug' | 'suggestion' | 'other' (required)
 *   - message: string, 10-1000 chars after trim (required)
 *   - version: string (optional, defaults to 'unknown')
 *
 * Returns:
 *   - 200: { success: true }
 *   - 400: { success: false, error: string } (validation)
 *   - 401: { success: false, error: 'Unauthorized' }
 *   - 429: { success: false, error: 'rate_limit' }
 *   - 500: { success: false, error: string } (server/email)
 *   - 504: { success: false, error: 'Email service timeout' }
 */

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2';
```

**CORS section** — copy `ALLOWED_ORIGINS` and `getCorsHeaders` exactly from `create-checkout/index.ts`:
- `ALLOWED_ORIGINS = ['https://testpianomaster.netlify.app', 'http://localhost:5174']`
- `getCorsHeaders(req)` returns origin-aware CORS headers with `POST, OPTIONS` methods and `authorization, x-client-info, apikey, content-type` allowed headers

**Deno.serve handler:**

1. `const corsHeaders = getCorsHeaders(req);`

2. Define `jsonResponse(body, status)` inside handler (same as create-checkout pattern — uses `corsHeaders` from closure)

3. **OPTIONS preflight:** Return `new Response(null, { status: 200, headers: corsHeaders })` (match send-consent-email style with `null` body)

4. **POST-only guard:** Return 405 if method is not POST (per D-03)

5. **JWT extraction** (per D-01, SPAM-01):
   - Get `Authorization` header; if missing, return `jsonResponse({ success: false, error: 'Unauthorized' }, 401)`
   - Create user client with `createClient(SUPABASE_URL, SUPABASE_ANON_KEY, { global: { headers: { Authorization: authHeader } } })`
   - Call `supabaseUser.auth.getUser()` — if error or no user, return 401
   - `user.id` is the authenticated student_id

6. **Parse + validate body** (per D-04, D-05, D-06, SPAM-04):
   - Parse `req.json()` — wrap in try/catch, return 400 on parse failure with `{ success: false, error: 'Invalid request body' }`
   - Extract `type`, `message`, `version` from body
   - `const VALID_TYPES = ['bug', 'suggestion', 'other'];`
   - If `!VALID_TYPES.includes(type)`, return `jsonResponse({ success: false, error: 'Invalid feedback type' }, 400)`
   - `const trimmed = (message ?? '').trim();`
   - If `trimmed.length < 10`, return `jsonResponse({ success: false, error: 'Message must be at least 10 characters' }, 400)`
   - If `trimmed.length > 1000`, return `jsonResponse({ success: false, error: 'Message must not exceed 1000 characters' }, 400)`

7. **Rate check** (per D-07, D-08, D-09, D-10, SPAM-02):
   - Create service role client: `createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY)` — service role bypasses RLS for the COUNT query
   - `const oneHourAgo = new Date(Date.now() - 60 * 60 * 1000).toISOString();`
   - Query: `supabaseService.from('feedback_submissions').select('*', { count: 'exact', head: true }).eq('student_id', user.id).gt('created_at', oneHourAgo)`
   - If `countError`, log and return 500
   - If `(count ?? 0) >= 3`, return `jsonResponse({ success: false, error: 'rate_limit' }, 429)` (per D-09)

8. **Insert submission row** (using service role client already open):
   - `supabaseService.from('feedback_submissions').insert({ student_id: user.id, type })`
   - If insert error, log and return 500
   - Note: message content is NOT stored in DB (per D-07 comment, COPPA-safe)

9. **Build email** (per D-11, D-12, D-13, D-14, D-15, D-16, BACK-02):
   - `const typeLabel = type.charAt(0).toUpperCase() + type.slice(1);` (e.g., "Bug", "Suggestion", "Other")
   - `const studentPrefix = user.id.substring(0, 8);` (per D-15: first 8 chars only)
   - `const timestamp = new Date().toISOString();`
   - `const appVersion = version ?? 'unknown';` (per D-18: missing version shows "unknown")
   - Subject: `` `[${typeLabel}] PianoMaster Feedback` `` (per D-12)
   - Email body (plain text, joined with `\n`):
     ```
     Type: ${typeLabel}
     Student: ${studentPrefix}...
     Version: ${appVersion}
     Submitted: ${timestamp}

     --- Message ---
     ${trimmed}
     ```

10. **Send via Brevo** (per BACK-01, D-14):
    - Read env vars: `BREVO_API_KEY`, `SENDER_EMAIL` (with fallback `'noreply@pianomaster.app'`), `SENDER_NAME` (with fallback `'PianoMaster'`)
    - If `!BREVO_API_KEY`, log error, return 500 with `{ success: false, error: 'Email service configuration error' }`
    - AbortController with 30-second timeout (same as send-consent-email)
    - `fetch('https://api.brevo.com/v3/smtp/email', { ... })` with:
      - `sender: { name: SENDER_NAME, email: SENDER_EMAIL }`
      - `to: [{ email: SENDER_EMAIL }]` (per D-14: feedback loops to same inbox)
      - `subject: [Type] PianoMaster Feedback`
      - `textContent: emailBody` (per D-11: plain text, NOT htmlContent)
    - `clearTimeout(timeoutId)` after response
    - If `!brevoResponse.ok`: log error with status + body, return `jsonResponse({ success: false, error: 'Email delivery failed' }, 500)`
    - If `AbortError`: return `jsonResponse({ success: false, error: 'Email service timeout' }, 504)`

11. **Success response:** `return jsonResponse({ success: true })`

12. **Outer try/catch:** Log error, return `jsonResponse({ success: false, error: 'Internal server error' }, 500)`

**Important anti-patterns to AVOID:**
- Do NOT import from shared modules — Edge Functions are self-contained (copy getCorsHeaders)
- Do NOT use anon client for rate check COUNT (no SELECT RLS policy per D-10, use service role)
- Do NOT send full student UUID in email (D-15: first 8 chars only)
- Do NOT reject missing version field (D-18: fallback to 'unknown')
- Do NOT use `htmlContent` for the Brevo call — use `textContent` (D-11: plain text)
