    **Module 1: `src/components/games/rhythm-games/utils/rhythmVexflowHelpers.js`**

    Converts binary patterns from RhythmPatternGenerator into renderable VexFlow notes.

    ```javascript
    import { StaveNote, Stem, Dot } from 'vexflow';

    // Map sixteenth-note duration units to VexFlow duration codes
    export const DURATION_TO_VEX = {
      16: 'w', 12: 'hd', 8: 'h', 6: 'qd', 4: 'q', 3: '8d', 2: '8', 1: '16'
    };

    // Convert binary array [1,0,0,0,1,0,...] to beat objects [{durationUnits, isRest}, ...]
    export function binaryPatternToBeats(binaryPattern) {
      const beats = [];
      let i = 0;
      while (i < binaryPattern.length) {
        const isNote = binaryPattern[i] === 1;
        let duration = 1;
        // Count consecutive 0s after this position
        let j = i + 1;
        while (j < binaryPattern.length && binaryPattern[j] === 0) {
          duration++;
          j++;
        }
        beats.push({ durationUnits: duration, isRest: !isNote });
        i = j;
      }
      return beats;
    }

    // Convert beat objects to VexFlow StaveNote objects for rhythm-only display
    export function beatsToVexNotes(beats) {
      return beats.map(beat => {
        const vexDur = DURATION_TO_VEX[beat.durationUnits];
        if (!vexDur) {
          // Split into representable durations if not in map (fallback to quarter)
          const fallbackDur = 'q';
          const note = new StaveNote({
            keys: ['b/4'],
            duration: beat.isRest ? fallbackDur + 'r' : fallbackDur,
            stem_direction: Stem.UP,
          });
          return note;
        }
        const isDotted = vexDur.endsWith('d');
        const baseDur = isDotted ? vexDur.slice(0, -1) : vexDur;
        const note = new StaveNote({
          keys: ['b/4'],
          duration: beat.isRest ? baseDur + 'r' : baseDur,
          stem_direction: Stem.UP,
        });
        if (isDotted) {
          Dot.buildAndAttach([note], { all: true });
        }
        return note;
      });
    }
    ```

    // Port from MetronomeTrainer calculateTimingThresholds (lines 47-64)
    // Base thresholds at 60 BPM, scaled proportionally by tempo
    export function calculateTimingThresholds(tempo) {
      const scale = 60 / tempo; // Slower tempo = wider window, faster = tighter
      return {
        PERFECT: 50 * scale,  // ms — 50ms at 60 BPM, 25ms at 120 BPM
        GOOD: 100 * scale,    // ms — 100ms at 60 BPM, 50ms at 120 BPM
        FAIR: 150 * scale,    // ms — 150ms at 60 BPM, 75ms at 120 BPM
      };
    }

    // Generate distractor patterns for dictation game (RDICT-04)
    // Each distractor differs from correct by exactly one duration swap
    export function generateDistractors(correctBeats, count = 2) {
      // ... detailed implementation:
      // 1. Find non-rest beats eligible for duration swap
      // 2. For each distractor: pick a different beat index, swap its duration
      //    with next-longer or next-shorter from SWAP_MAP
      // 3. Compensate: adjust adjacent beat or add/modify rest to maintain measure length
      // 4. Validate total durationUnits sum equals original
      // Return array of count distractor beat arrays
    }

    // Schedule audio playback of a pattern via playNote callback
    export function schedulePatternPlayback(beats, tempo, audioContext, playNote) {
      const beatDuration = 60 / tempo; // seconds per quarter note
      const sixteenthDuration = beatDuration / 4; // seconds per sixteenth note
      const startTime = audioContext.currentTime + 0.1; // small buffer
      let offset = 0;
      beats.forEach(beat => {
        if (!beat.isRest) {
          playNote('C4', { startTime: startTime + offset, duration: sixteenthDuration * beat.durationUnits * 0.8 });
        }
        offset += sixteenthDuration * beat.durationUnits;
      });
      return { startTime, totalDuration: offset };
    }
