       ```javascript
       function updateCursor() {
         const ctx = audioContextRef.current;
         if (!ctx) return;
         const elapsed = ctx.currentTime - patternStartTimeRef.current;
         const totalDuration = measureDurationSeconds; // (60/tempo) * beats
         const progress = Math.min(elapsed / totalDuration, 1);
         if (cursorDivRef.current) {
           cursorDivRef.current.style.left = `${progress * 100}%`;
         }
         if (progress < 1) {
           rafIdRef.current = requestAnimationFrame(updateCursor);
         } else {
           // Pattern complete — transition to FEEDBACK
         }
       }
       ```
