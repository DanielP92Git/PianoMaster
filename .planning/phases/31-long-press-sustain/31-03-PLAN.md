const handlePressStart = useCallback(
  (e) => {
    if (phase !== PHASES.PLAYING) return;

    if (!currentBeatIsHold) {
      // Quarter note -- use existing tap logic
      handleTap();
      return;
    }

    // Hold note
    pressStartTimeRef.current = performance.now();
    setIsHoldComplete(false);
    setHoldFeedbackLabel(null);

    // Sustained piano sound
    const noteDurationSec = currentHoldDurationMs / 1000;
    audioEngine.createPianoSound(
      audioEngine.getCurrentTime(),
      0.8,
      noteDurationSec
    );

    // Record tap time for onset accuracy (same as existing handleTap)
    const tapTime = audioEngine.getCurrentTime();
    userTapsRef.current.push({ time: tapTime });

    // Start rAF ring animation
    const startTime = performance.now();
    const requiredMs = currentHoldDurationMs;
    const animate = () => {
      const elapsed = performance.now() - startTime;
      const progress = Math.min(1, elapsed / requiredMs);
      if (holdRingCircleRef.current) {
        const offset = CIRCUMFERENCE * (1 - progress);
        holdRingCircleRef.current.setAttribute(
          "stroke-dashoffset",
          String(offset)
        );
      }
      if (progress < 1 && pressStartTimeRef.current !== null) {
        rafIdRef.current = requestAnimationFrame(animate);
      }
    };
    rafIdRef.current = requestAnimationFrame(animate);

    // Trigger visual flash (same as existing handleTap beat flash)
    setBeatPulse(true);
  },
  [
    phase,
    currentBeatIsHold,
    currentHoldDurationMs,
    handleTap,
    audioEngine,
    tempo,
  ]
);

const handlePressEnd = useCallback(() => {
  if (pressStartTimeRef.current === null) return;

  cancelAnimationFrame(rafIdRef.current);
  const holdMs = performance.now() - pressStartTimeRef.current;
  pressStartTimeRef.current = null;

  const quality = scoreHold(holdMs, currentHoldDurationMs);

  if (quality === "PERFECT") {
    setIsHoldComplete(true);
    setTimeout(() => setIsHoldComplete(false), 200);

  if (holdRingCircleRef.current) {
    holdRingCircleRef.current.setAttribute(
      "stroke-dashoffset",
      String(CIRCUMFERENCE)
    );
  }

  // Show feedback
  if (quality === "GOOD") {
    setHoldFeedbackLabel(t("games.metronomeTrainer.tapArea.accuracy.holdGood"));

**Step 7 -- Add stretched beat indicator (D-08):**
The `MetronomeDisplay` component renders equal-width circles for each beat. For hold notes, D-08 requires beat indicators that span multiple positions.

Create a custom beat indicator row BELOW or INSTEAD OF the standard MetronomeDisplay when the beats array contains hold notes:

```jsx
{
  /* Stretched beat indicator for hold notes -- only renders when beats contain non-quarter notes */
}
{
  beats.some((b) => b.durationUnits > 4) && (
    <div
      className="flex justify-center gap-1"
      dir="ltr"
      style={{
        display: "grid",
        gridTemplateColumns: `repeat(${beats.reduce((sum, b) => sum + b.durationUnits / 4, 0)}, 1fr)`,
        gap: "4px",
        maxWidth: "200px",
        margin: "0 auto",
      }}
    >
      {beats.map((beat, i) => {
        const span = beat.durationUnits / 4; // quarter=1, half=2, whole=4
        const isCurrent = (currentBeat - 1) % beats.length === i;
        return (
          <div
            key={i}
            style={{ gridColumn: `span ${span}` }}
            className={`h-3 rounded-full transition-all duration-150 ${
              isCurrent ? "scale-110 bg-blue-400" : "bg-white/20"
            }`}
          />
        );
      })}
    </div>
  );
}
```

When ALL beats are quarters (the default `PULSE_BEATS`), this stretched indicator does not render -- the standard `MetronomeDisplay` handles it (no visual change for default pulse exercises).

When beats contain half or whole notes, both the stretched indicator AND MetronomeDisplay can coexist, or the stretched indicator replaces MetronomeDisplay. The executor should evaluate which looks better. The stretched indicator is the required one (D-08); MetronomeDisplay is supplementary.
