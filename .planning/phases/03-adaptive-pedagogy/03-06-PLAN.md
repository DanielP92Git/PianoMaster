```javascript
// Phase 03 (ADAPT-03/D-09/D-11): bias the baseline pool toward historically-weak pitches,
// strictly within the node's own notePool. Cold start / free-play → uniform (no-op).
let cancelled = false;
(async () => {
  if (!nodeId || !studentId) return;
  try {
    const progress = await getNodeProgress(studentId, nodeId); // verifyStudentDataAccess gate inside
    const masteryMap = progress?.note_mastery || {};
    if (cancelled) return;
    const baseNotes = builtSettings.selectedNotes || [];
    const weighted = buildWeightedNotePool(baseNotes, masteryMap, MASTERY_MIN_ATTEMPTS);
    // Seed both the pristine adaptive baseline AND current settings so exercise 1 is already biased.
    baseAdaptiveSettingsRef.current = { ...baseAdaptiveSettingsRef.current, selectedNotes: weighted };
    setGameSettings((s) => ({ ...s, selectedNotes: weighted }));
  } catch (e) {
    // Non-fatal: fall back to uniform baseline (no biasing). Log via existing error path only.
  }
})();
return () => { cancelled = true; };
```
