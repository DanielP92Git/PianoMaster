    ### scripts/validateTrail.mjs — validateMeasureCountPolicy (D-15, PAT-06)

    1. Add a new function `validateMeasureCountPolicy()` after the existing `validateGameTypePolicy()`:

    ```javascript
    /**
     * Validate that each rhythm node's measureCount matches its nodeType policy.
     * Policy (from Phase 23 D-12):
     *   discovery=1, practice=2, mix_up=1, review=2,
     *   challenge=2, speed_round=4, mini_boss=4, boss=4
     */
    function validateMeasureCountPolicy() {
      const MEASURE_COUNT_POLICY = {
        discovery: 1,
        practice: 2,
        mix_up: 1,
        review: 2,
        challenge: 2,
        speed_round: 4,
        mini_boss: 4,
        boss: 4,
      };

      const errors = [];

      for (const node of SKILL_NODES) {
        if (node.category !== 'rhythm') continue;
        if (!node.nodeType) continue;

        const expectedMeasureCount = MEASURE_COUNT_POLICY[node.nodeType];
        if (expectedMeasureCount === undefined) continue;

        // Check each exercise's config.measureCount
        if (node.exercises) {
          for (let i = 0; i < node.exercises.length; i++) {
            const ex = node.exercises[i];
            const mc = ex.config?.measureCount;
            // Skip exercise types that don't use measureCount (e.g., pulse, visual_recognition, syllable_matching)
            if (['pulse', 'visual_recognition', 'syllable_matching', 'mixed_lesson', 'note_recognition', 'memory_game'].includes(ex.type)) continue;
            if (mc !== undefined && mc !== expectedMeasureCount) {
              errors.push(`${node.id} exercise[${i}] (${ex.type}): measureCount=${mc}, expected=${expectedMeasureCount} for nodeType=${node.nodeType}`);
            }
          }
        }
      }

      if (errors.length > 0) {
        console.error('\n--- Measure Count Policy Violations ---');
        errors.forEach(e => console.error(`  ERROR: ${e}`));
        process.exitCode = 1;
      } else {
        console.log('Measure count policy: PASS');
      }
    }
    ```
