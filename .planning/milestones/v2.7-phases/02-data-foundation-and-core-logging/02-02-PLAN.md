**handleLog function:**
```javascript
const handleLog = useCallback(() => {
  if (logState !== 'idle') return;
  setLogState('logging');

  logMutation.mutate(localDate, {
    onSuccess: async ({ inserted }) => {
      if (inserted) {
        // Update streak
        const weekendPass = queryClient.getQueryData(['streak-state', user?.id]);
        const wpEnabled = weekendPass?.weekendPassEnabled ?? false;
        await practiceStreakService.updatePracticeStreak(localDate, wpEnabled);
        queryClient.invalidateQueries({ queryKey: ['practice-streak', user?.id] });
        queryClient.invalidateQueries({ queryKey: ['xp'] }); // refresh XP display
      }
      // Instant cache update (no loading flash on return)
      queryClient.setQueryData(['practice-log-today', user?.id, localDate], { logged: true });

      // 2-second hold before settling (D-07)
      setTimeout(() => setLogState('settled'), 2000);
    },
    onError: () => {
      // Silent failure — revert to idle (D-07 error state)
      setLogState('idle');
    },
  });
}, [logState, localDate, user?.id, logMutation, queryClient]);
```

**On mount: if logStatus.logged === true, initialize logState to 'settled'.**

Use `useEffect` to set `logState` to `'settled'` when `logStatus?.logged === true` and `logState === 'idle'` (handles returning to dashboard after logging).
