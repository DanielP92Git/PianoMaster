Root cause: `binaryToVexDurations` generates rest codes (qr, hr, wr) via `REST_EQUIVALENT` whenever a gap between onsets exceeds the largest available note duration. Early-curriculum nodes (Units 1-3) have `durations` that include only note codes (no rest codes), yet the resolver still produces rests.

Fix approach: Add `allowRests` option to `resolveByTags`. When false, pre-filter patterns to only those whose gaps are coverable by the available note durations without rests. This is a pattern-selection filter, NOT a modification to `binaryToVexDurations` (which correctly renders what it receives).

**Implementation steps:**

1. Add a helper function `patternNeedsRests(binary, durations)` that returns `true` if rendering the binary with the given durations would require any rest codes:

```javascript
/**
 * Check whether a binary pattern would require rest codes when rendered
 * with the given durations. Returns true if ANY gap (leading, between onsets,
 * or trailing) exceeds the longest available note duration.
 */
function patternNeedsRests(binary, durations) {
  const maxNoteSlots = Math.max(
    ...durations
      .filter(
        (d) =>
          DURATION_SLOTS[d] !== undefined &&
          !REST_EQUIVALENT[d]?.startsWith?.("skip")
      )
      .filter((d) => !d.endsWith("r")) // exclude rest codes from duration list
      .map((d) => DURATION_SLOTS[d]),
    0
  );
  if (maxNoteSlots === 0) return true; // No note durations available

  const onsets = [];
  for (let i = 0; i < binary.length; i++) {
    if (binary[i] === 1) onsets.push(i);
  }
  if (onsets.length === 0) return true; // No onsets = all rests

  // Check leading gap (before first onset)
  if (onsets[0] > 0) return true;

  // Check gaps between consecutive onsets
  for (let i = 0; i < onsets.length; i++) {
    const nextOnset = i + 1 < onsets.length ? onsets[i + 1] : binary.length;
    const gap = nextOnset - onsets[i];
    // The gap must be fillable by available note durations.
    // Use greedy fill: subtract largest fitting duration repeatedly.
    let remaining = gap;
    while (remaining > 0) {
      // Find largest duration that fits
      const fit = durations
        .filter((d) => !d.endsWith("r") && DURATION_SLOTS[d] !== undefined)
        .map((d) => DURATION_SLOTS[d])
        .filter((s) => s <= remaining)
        .sort((a, b) => b - a)[0];
      if (!fit) return true; // No note fits remaining gap → needs rest
      remaining -= fit;
    }
  }

  return false;
}
```

2. Update `resolveByTags` to accept `options.allowRests` (boolean, default `false`):

```javascript
export function resolveByTags(tags, durations, options = {}) {
  const { allowRests = false, timeSignature: tsFilter } = options;

  let matching = RHYTHM_PATTERNS.filter((p) =>
    tags.every((tag) => p.tags.includes(tag))
  );

  if (tsFilter) {
    matching = matching.filter((p) => p.timeSignature === tsFilter);
  }

  // When rests are not allowed, filter out patterns that would produce rest codes
  if (!allowRests) {
    matching = matching.filter((p) => !patternNeedsRests(p.pattern, durations));
  }

  if (matching.length === 0) return null;

  const selected = matching[Math.floor(Math.random() * matching.length)];
  const vexDurations = binaryToVexDurations(
    selected.pattern,
    durations,
    selected.timeSignature
  );

  return {
    patternId: selected.id,
    binary: selected.pattern,
    timeSignature: selected.timeSignature,
    vexDurations,
    tags: selected.tags,
  };
}
```
