    **Step 2.1 — Create `src/components/games/rhythm-games/components/BossIntroOverlay.jsx`:**

    ```jsx
    import { useEffect, useState } from "react";
    import { motion, AnimatePresence } from "framer-motion";
    import { useTranslation } from "react-i18next";
    import { Crown } from "lucide-react";
    import { useMotionTokens } from "../../../../utils/useMotionTokens";
    import { useAccessibility } from "../../../../contexts/AccessibilityContext";

    /**
     * Boss intro overlay (D-18, Phase 33 Plan 33-08).
     * Mounts before the first question of a full BOSS node, displays for 2 seconds,
     * then dismisses via onDismiss. Respects reducedMotion + i18n EN/HE.
     *
     * Per CONTEXT D-18: intro overlay + victory VFX, NO new mechanics, NO music swap.
     */
    export function BossIntroOverlay({ bossName, onDismiss }) {
      const { t, i18n } = useTranslation("trail");
      const { fade } = useMotionTokens();
      const { reducedMotion } = useAccessibility();
      const isRTL = i18n.dir() === "rtl";
      const [visible, setVisible] = useState(true);

      useEffect(() => {
        const t = setTimeout(() => {
          setVisible(false);
          onDismiss?.();
        }, 2000);
        return () => clearTimeout(t);
      }, [onDismiss]);

      return (
        <AnimatePresence>
          {visible && (
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              transition={fade}
              dir={isRTL ? "rtl" : "ltr"}
              role="status"
              aria-live="polite"
              aria-atomic="true"
              className="fixed inset-0 z-[9999] flex flex-col items-center justify-center gap-4 bg-gradient-to-br from-amber-900/95 via-orange-900/95 to-red-900/95 backdrop-blur-md px-4"
            >
              <Crown className="w-16 h-16 text-amber-300" aria-hidden="true" />
              {reducedMotion ? (
                <h1 className="text-5xl font-extrabold text-amber-300 select-none" style={{ fontFamily: "Fredoka One, sans-serif" }}>
                  {t("boss.intro.title", "BOSS FIGHT")}
                </h1>
              ) : (
                <motion.h1
                  initial={{ scale: 0.5 }}
                  animate={{ scale: [0.5, 1.1, 1] }}
                  transition={{ duration: 0.6 }}
                  className="text-5xl font-extrabold text-amber-300 select-none"
                  style={{ fontFamily: "Fredoka One, sans-serif" }}
                >
                  {t("boss.intro.title", "BOSS FIGHT")}
                </motion.h1>
              )}
              {bossName && (
                <p className="text-xl text-white/90">{bossName}</p>
              )}
            </motion.div>
          )}
        </AnimatePresence>
      );
    }
    ```
